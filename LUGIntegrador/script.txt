-- Crear tablas
CREATE TABLE AnalistaDeportivo (
    Id BIGINT PRIMARY KEY,
    Nombre NVARCHAR(100),
    Apellido NVARCHAR(100),
    DNI NVARCHAR(50),
    Telefono NVARCHAR(50),
    FechaNacimiento DATE,
    Matricula NVARCHAR(50)
);

CREATE TABLE Entrenador (
    Id BIGINT PRIMARY KEY,
    Nombre NVARCHAR(100),
    Apellido NVARCHAR(100),
    DNI NVARCHAR(50),
    Telefono NVARCHAR(50),
    FechaNacimiento DATE,
    NumeroLicencia NVARCHAR(50),
    Titulo NVARCHAR(100)
);

CREATE TABLE Equipo (
    Id BIGINT PRIMARY KEY,
    Nombre NVARCHAR(100),
    Descripcion NVARCHAR(255)
);

CREATE TABLE Jugador (
    Id BIGINT PRIMARY KEY,
    Nombre NVARCHAR(100),
    Apellido NVARCHAR(100),
    DNI NVARCHAR(50),
    Telefono NVARCHAR(50),
    FechaNacimiento DATE,
    Posicion NVARCHAR(50),
    Edad INT,
    Altura INT,
    Peso INT,
    EquipoId BIGINT,
    FOREIGN KEY (EquipoId) REFERENCES Equipo(Id)
);

CREATE TABLE Campeonato (
    Id BIGINT PRIMARY KEY,
    Nombre NVARCHAR(100),
    FechaInicio DATE,
    FechaFin DATE,
    CantidadPartidos INT,
    CantidadJugadores INT
);

CREATE TABLE Partido (
    Id BIGINT PRIMARY KEY,
    Fecha DATE,
    Duracion INT,
    NumeroCancha INT,
    Ubicacion NVARCHAR(255),
    Categoria NVARCHAR(100),
    CampeonatoId BIGINT,
    FOREIGN KEY (CampeonatoId) REFERENCES Campeonato(Id)
);

CREATE TABLE Convocatoria (
    Id BIGINT PRIMARY KEY,
    Posicion NVARCHAR(50),
    Confirmacion BIT,
    Fecha DATE,
    Duracion TIME,
    Ubicacion NVARCHAR(255),
    JugadorId BIGINT,
    PartidoId BIGINT,
    FOREIGN KEY (JugadorId) REFERENCES Jugador(Id),
    FOREIGN KEY (PartidoId) REFERENCES Partido(Id)
);

CREATE TABLE Estadistica (
    Id BIGINT PRIMARY KEY,
    PasesConectados INT,
    PasesFallados INT,
    Asistencias INT,
    KmRecorrido INT,
    TarjetaRoja BIT,
    TarjetaAmarilla BIT,
    DuelosGanados INT,
    DuelosPerdidos INT,
    Bloqueos INT,
    Disparos INT,
    Goles INT,
    Recuperaciones INT,
    JugadorId BIGINT,
    PartidoId BIGINT,
    FOREIGN KEY (JugadorId) REFERENCES Jugador(Id),
    FOREIGN KEY (PartidoId) REFERENCES Partido(Id)
);
